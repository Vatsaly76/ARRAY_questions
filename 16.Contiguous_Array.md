## âœ… Optimized Solution (Prefix Sum + HashMap)

```js
var findMaxLength = function(nums) {
    let maxLength = 0;
    let sum = 0;
    let map = new Map();
    map.set(0, -1); // base case: sum 0 at index -1

    for (let i = 0; i < nums.length; i++) {
        sum += nums[i] === 0 ? -1 : 1;

        if (map.has(sum)) {
            maxLength = Math.max(maxLength, i - map.get(sum));
        } else {
            map.set(sum, i);
        }
    }

    return maxLength;
};
```
